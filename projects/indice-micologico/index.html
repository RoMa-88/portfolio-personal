🧮 Algoritmo del Índice

def indice_micologico_simple(lluvia7, lluvia14, pct_t_ok): # Normalizaciones: 20mm/7d = "ple", 40mm/14d = "ple" n7 = min(lluvia7 / 20.0, 1.0) n14 = min(lluvia14 / 40.0, 1.0) return 0.5*n7 + 0.3*n14 + 0.2*pct_t_ok
